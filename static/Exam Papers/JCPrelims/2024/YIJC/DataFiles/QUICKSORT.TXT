FUNCTION QuickSort(seq)
  QuickSortHelper(seq, 0, LENGTH(seq) - 1)
  RETURN seq  
ENDFUNCTION

FUNCTION QuickSortHelper(seq, start, end)
  IF start < end 
    THEN
      mid ← Partition(seq, start, end) 
      QuickSortHelper(seq, start, mid - 1)
      QuickSortHelper(seq, mid + 1, end)
  ENDIF
  RETURN seq  
ENDFUNCTION
